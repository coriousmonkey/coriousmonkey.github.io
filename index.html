Jalankan query berikut : 

select * from
        (
     		
     select absensi.id,absensi.`timestamp`, 
absensi.pertemuan as `pertemuanke`,
absensi.ket,

mahasiswa.nim,
mahasiswa.`user`,
mahasiswa.angkatan,
mahasiswa.kelas

from absensi
left join mahasiswa
on absensi.mahasiswa=mahasiswa.id
where absensi.jadwal=1
order by mahasiswa.id
     
        )
        as tbb
        group by nim,pertemuanke
        order by pertemuanke,nim
